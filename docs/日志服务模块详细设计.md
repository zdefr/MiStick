## 2. 模块概述

### 2.1 职责

日志服务模块负责应用日志的记录、存储、查询和导出，采用**SQLite + JSONL 归档**的分级存储策略，支持高效查询和统计分析。

### 2.2 功能范围

| 功能 | 说明 | 优先级 |
|------|------|--------|
| 日志记录 | 记录各级别日志到 SQLite | P0 |
| 日志分级 | DEBUG/INFO/WARN/ERROR | P0 |
| 日志查询 | SQL 条件查询，支持索引优化 | P0 |
| 日志归档 | 7 天前日志自动归档到 JSONL | P1 |
| 日志导出 | 导出日志为 JSONL/CSV | P1 |
| 日志统计 | 按级别/模块聚合统计 | P1 |
| 日志清理 | 自动清理过期归档 | P1 |

### 2.3 设计变更说明

**v1.1 变更：** 从纯 JSON 文件存储改为 SQLite + JSONL 分级存储

| 变更项 | 原设计 | 新设计 | 理由 |
|--------|--------|--------|------|
| 存储介质 | JSON 文件 | SQLite + JSONL | 查询性能、并发安全 |
| 查询方式 | 线性扫描 | SQL 索引查询 | 大数据量性能 |
| 归档策略 | 无 | 冷热分离 | 控制数据库大小 |
| 统计功能 | 不支持 | SQL 聚合 | 运维分析需求 |

### 4.2 核心类说明

#### 4.2.1 Logger

日志服务主类，单例模式。

```typescript
import Database from 'better-sqlite3';
import { EventEmitter } from 'events';
import fs from 'fs/promises';
import path from 'path';
import { app } from 'electron';
import zlib from 'zlib';

enum LogLevel {
  DEBUG = 0,
  INFO = 1,
  WARN = 2,
  ERROR = 3,
}

interface LogEntry {
  id?: number;
  timestamp: string;
  level: string;
  module: string;
  message: string;
  meta?: any;
  stack?: string;
}

interface LogQueryOptions {
  level?: LogLevel;
  module?: string;
  startTime?: Date;
  endTime?: Date;
  limit?: number;
  offset?: number;
  search?: string;
}

interface LogStats {
  total: number;
  byLevel: Record<string, number>;
  byModule: Record<string, number>;
  errorRate: number;
}

class Logger extends EventEmitter {
  private static instance: Logger | null = null;
  private db: Database;
  private logDir: string;
  private archiveDir: string;
  private retentionDays: number = 7;
  private archiveRetentionDays: number = 30;
  private level: LogLevel = LogLevel.INFO;
  
  // 批量写入缓冲
  private writeBuffer: LogEntry[] = [];
  private flushTimer: NodeJS.Timeout | null = null;
  private readonly BUFFER_SIZE = 100;
  private readonly FLUSH_INTERVAL = 1000;  // 1 秒

  private constructor() {
    super();
    this.logDir = path.join(app.getPath('userData'), 'logs');
    this.archiveDir = path.join(this.logDir, 'archive');
    
    // 初始化数据库
    this.db = new Database(path.join(app.getPath('userData'), 'logs.db'));
    this.initDatabase();
    
    // 确保目录存在
    this.ensureDirectories();
    
    // 启动定时任务
    this.startScheduledTasks();
  }

  static getInstance(): Logger {
    if (!Logger.instance) {
      Logger.instance = new Logger();
    }
    return Logger.instance;
  }

  /**
   * 初始化数据库表
   */
  private initDatabase(): void {
    this.db.exec(`
      CREATE TABLE IF NOT EXISTS logs (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        timestamp TEXT NOT NULL,
        level TEXT NOT NULL CHECK(level IN ('DEBUG', 'INFO', 'WARN', 'ERROR')),
        module TEXT NOT NULL,
        message TEXT NOT NULL,
        meta TEXT,
        stack TEXT
      );
      
      CREATE INDEX IF NOT EXISTS idx_timestamp ON logs(timestamp);
      CREATE INDEX IF NOT EXISTS idx_level ON logs(level);
      CREATE INDEX IF NOT EXISTS idx_module ON logs(module);
    `);
  }

  /**
   * 确保目录存在
   */
  private async ensureDirectories(): Promise<void> {
    await fs.mkdir(this.logDir, { recursive: true });
    await fs.mkdir(this.archiveDir, { recursive: true });
  }

  /**
   * 启动定时任务
   */
  private startScheduledTasks(): void {
    // 每 5 分钟检查是否需要归档
    setInterval(() => this.archiveOldLogs(), 5 * 60 * 1000);
    
    // 每天清理过期归档
    setInterval(() => this.cleanupOldArchives(), 24 * 60 * 60 * 1000);
  }

  /**
   * 调试日志
   */
  debug(message: string, meta?: any): void {
    this.log(LogLevel.DEBUG, message, meta);
  }

  /**
   * 信息日志
   */
  info(message: string, meta?: any): void {
    this.log(LogLevel.INFO, message, meta);
  }

  /**
   * 警告日志
   */
  warn(message: string, meta?: any): void {
    this.log(LogLevel.WARN, message, meta);
  }

  /**
   * 错误日志
   */
  error(message: string, meta?: any): void {
    this.log(LogLevel.ERROR, message, meta);
  }

  /**
   * 设置日志级别
   */
  setLevel(level: LogLevel): void {
    this.level = level;
  }

  /**
   * 核心日志方法
   */
  private log(level: LogLevel, message: string, meta?: any): void {
    if (level < this.level) {
      return;
    }

    const entry: LogEntry = {
      timestamp: new Date().toISOString(),
      level: LogLevel[level],
      module: this.getCallerModule(),
      message,
      meta: meta instanceof Error ? { message: meta.message } : meta,
    };

    // 错误日志添加堆栈跟踪
    if (level === LogLevel.ERROR && meta instanceof Error) {
      entry.stack = meta.stack;
    }

    // 添加到缓冲区
    this.writeBuffer.push(entry);

    // 检查是否需要刷新
    if (this.writeBuffer.length >= this.BUFFER_SIZE) {
      this.flush();
    } else if (!this.flushTimer) {
      this.flushTimer = setTimeout(() => this.flush(), this.FLUSH_INTERVAL);
    }
  }

  /**
   * 刷新缓冲区到数据库
   */
  private flush(): void {
    if (this.writeBuffer.length === 0) return;

    if (this.flushTimer) {
      clearTimeout(this.flushTimer);
      this.flushTimer = null;
    }

    const insert = this.db.prepare(`
      INSERT INTO logs (timestamp, level, module, message, meta, stack)
      VALUES (?, ?, ?, ?, ?, ?)
    `);

    const insertMany = this.db.transaction((entries: LogEntry[]) => {
      for (const entry of entries) {
        insert.run(
          entry.timestamp,
          entry.level,
          entry.module,
          entry.message,
          entry.meta ? JSON.stringify(entry.meta) : null,
          entry.stack || null
        );
      }
    });

    try {
      insertMany(this.writeBuffer);
      this.writeBuffer = [];
    } catch (error) {
      console.error('Failed to flush logs:', error);
    }
  }

  /**
   * 查询日志
   */
  async query(options: LogQueryOptions): Promise<LogEntry[]> {
    const {
      level,
      module,
      startTime,
      endTime,
      limit = 100,
      offset = 0,
      search,
    } = options;

    const conditions: string[] = [];
    const params: any[] = [];

    if (level !== undefined) {
      conditions.push('level = ?');
      params.push(LogLevel[level]);
    }
    if (module !== undefined) {
      conditions.push('module = ?');
      params.push(module);
    }
    if (startTime) {
      conditions.push('timestamp >= ?');
      params.push(startTime.toISOString());
    }
    if (endTime) {
      conditions.push('timestamp <= ?');
      params.push(endTime.toISOString());
    }
    if (search) {
      conditions.push('message LIKE ?');
      params.push(`%${search}%`);
    }

    const where = conditions.length > 0 ? `WHERE ${conditions.join(' AND ')}` : '';

    const rows = this.db.prepare(`
      SELECT * FROM logs 
      ${where}
      ORDER BY timestamp DESC
      LIMIT ? OFFSET ?
    `).all(...params, limit, offset);

    return rows.map(row => ({
      ...row,
      meta: row.meta ? JSON.parse(row.meta) : null,
    }));
  }

  /**
   * 获取统计信息
   */
  getStats(startTime: Date, endTime: Date): LogStats {
    const timeRange = 'timestamp BETWEEN ? AND ?';
    
    // 总数
    const totalRow = this.db.prepare(`
      SELECT COUNT(*) as count FROM logs WHERE ${timeRange}
    `).get(startTime.toISOString(), endTime.toISOString()) as { count: number };

    // 按级别统计
    const byLevelRows = this.db.prepare(`
      SELECT level, COUNT(*) as count 
      FROM logs 
      WHERE ${timeRange}
      GROUP BY level
    `).all(startTime.toISOString(), endTime.toISOString()) as Array<{ level: string; count: number }>;

    // 按模块统计
    const byModuleRows = this.db.prepare(`
      SELECT module, COUNT(*) as count 
      FROM logs 
      WHERE ${timeRange}
      GROUP BY module
      LIMIT 20
    `).all(startTime.toISOString(), endTime.toISOString()) as Array<{ module: string; count: number }>;

    const byLevel: Record<string, number> = {};
    byLevelRows.forEach(row => { byLevel[row.level] = row.count; });

    const byModule: Record<string, number> = {};
    byModuleRows.forEach(row => { byModule[row.module] = row.count; });

    const errorCount = byLevel['ERROR'] || 0;
    const errorRate = totalRow.count > 0 ? (errorCount / totalRow.count) * 100 : 0;

    return {
      total: totalRow.count,
      byLevel,
      byModule,
      errorRate: parseFloat(errorRate.toFixed(2)),
    };
  }

  /**
   * 归档旧日志
   */
  private async archiveOldLogs(): Promise<void> {
    const cutoffDate = new Date();
    cutoffDate.setDate(cutoffDate.getDate() - this.retentionDays);

    // 查询旧日志
    const oldLogs = this.db.prepare(`
      SELECT * FROM logs WHERE timestamp < ?
    `).all(cutoffDate.toISOString()) as LogEntry[];

    if (oldLogs.length === 0) return;

    // 写入归档文件
    const archiveFile = path.join(
      this.archiveDir,
      `${cutoffDate.toISOString().split('T')[0]}.jsonl`
    );

    const content = oldLogs.map(log => JSON.stringify(log)).join('\n') + '\n';
    await fs.appendFile(archiveFile, content);

    // 压缩归档文件
    await this.compressFile(archiveFile);

    // 从数据库删除
    this.db.prepare(`
      DELETE FROM logs WHERE timestamp < ?
    `).run(cutoffDate.toISOString());
  }

  /**
   * 压缩文件
   */
  private async compressFile(filePath: string): Promise<void> {
    const content = await fs.readFile(filePath);
    const compressed = zlib.gzipSync(content);
    await fs.writeFile(filePath + '.gz', compressed);
    await fs.unlink(filePath);
  }

  /**
   * 清理过期归档
   */
  private async cleanupOldArchives(): Promise<void> {
    const cutoffDate = new Date();
    cutoffDate.setDate(cutoffDate.getDate() - this.archiveRetentionDays);

    const files = await fs.readdir(this.archiveDir);
    for (const file of files) {
      const dateStr = file.split('.')[0];
      const fileDate = new Date(dateStr);
      if (fileDate < cutoffDate) {
        await fs.unlink(path.join(this.archiveDir, file));
      }
    }
  }

  /**
   * 导出日志
   */
  async export(filePath: string, format: 'jsonl' | 'csv' = 'jsonl'): Promise<void> {
    const allLogs: LogEntry[] = [];
    
    // 从 SQLite 读取
    const chunkSize = 1000;
    let offset = 0;
    while (true) {
      const rows = this.db.prepare(`
        SELECT * FROM logs 
        ORDER BY timestamp DESC
        LIMIT ? OFFSET ?
      `).all(chunkSize, offset) as any[];
      
      if (rows.length === 0) break;
      
      allLogs.push(...rows.map(row => ({
        ...row,
        meta: row.meta ? JSON.parse(row.meta) : null,
      })));
      
      offset += chunkSize;
    }

    // 从归档文件读取
    const archiveFiles = await fs.readdir(this.archiveDir);
    for (const file of archiveFiles) {
      const fullPath = path.join(this.archiveDir, file);
      let content: string;
      
      if (file.endsWith('.gz')) {
        const compressed = await fs.readFile(fullPath);
        content = zlib.gunzipSync(compressed).toString();
      } else {
        content = await fs.readFile(fullPath, 'utf-8');
      }
      
      const lines = content.split('\n').filter(line => line.trim());
      allLogs.push(...lines.map(line => JSON.parse(line)));
    }

    // 写入导出文件
    if (format === 'jsonl') {
      const content = allLogs.map(log => JSON.stringify(log)).join('\n');
      await fs.writeFile(filePath, content);
    } else if (format === 'csv') {
      const headers = ['timestamp', 'level', 'module', 'message', 'meta', 'stack'];
      const rows = allLogs.map(log => 
        headers.map(h => {
          const val = log[h];
          return typeof val === 'object' ? JSON.stringify(val) : String(val || '');
        }).join(',')
      );
      await fs.writeFile(filePath, [headers.join(','), ...rows].join('\n'));
    }
  }

  /**
   * 清空日志
   */
  async clear(): Promise<void> {
    this.db.exec('DELETE FROM logs');
  }

  /**
   * 获取调用模块
   */
  private getCallerModule(): string {
    const error = new Error();
    const stack = error.stack?.split('\n') || [];
    const callerLine = stack[4];  // log -> getCallerModule -> caller -> ...
    const match = callerLine?.match(/docs[\\/](.+?)\.ts/);
    return match ? match[1] : 'unknown';
  }

  /**
   * 关闭数据库连接
   */
  destroy(): void {
    this.flush();
    if (this.flushTimer) {
      clearTimeout(this.flushTimer);
    }
    this.db.close();
  }
}
```
