## 2. 模块概述

### 2.1 职责

日志服务模块负责应用日志的记录、存储、查询和导出，支持分级日志和日志轮转。

### 2.2 功能范围

| 功能 | 说明 | 优先级 |
|------|------|--------|
| 日志记录 | 记录各级别日志 | P0 |
| 日志分级 | DEBUG/INFO/WARN/ERROR | P0 |
| 日志存储 | 文件存储，支持轮转 | P0 |
| 日志查询 | 按条件查询日志 | P1 |
| 日志导出 | 导出日志文件 | P1 |
| 日志清理 | 自动清理旧日志 | P1 |

### 3.2 核心类说明

#### 3.2.1 Logger

日志服务主类，单例模式。

```typescript
enum LogLevel {
  DEBUG = 0,
  INFO = 1,
  WARN = 2,
  ERROR = 3,
}

class Logger {
  private static instance: Logger | null = null;
  private logDir: string;
  private level: LogLevel;
  private maxFiles: number;
  private maxSize: number;  // bytes
  private currentFile: string;
  private currentSize: number;
  private writeStream: fs.WriteStream | null = null;
  
  private constructor() {
    this.logDir = path.join(app.getPath('userData'), 'logs');
    this.level = LogLevel.INFO;
    this.maxFiles = 7;
    this.maxSize = 10 * 1024 * 1024;  // 10MB
    
    this.ensureLogDir();
    this.rotateLog();
  }
  
  static getInstance(): Logger {
    if (!Logger.instance) {
      Logger.instance = new Logger();
    }
    return Logger.instance;
  }
  
  /**
   * 调试日志
   */
  debug(message: string, meta?: any): void {
    this.log(LogLevel.DEBUG, message, meta);
  }
  
  /**
   * 信息日志
   */
  info(message: string, meta?: any): void {
    this.log(LogLevel.INFO, message, meta);
  }
  
  /**
   * 警告日志
   */
  warn(message: string, meta?: any): void {
    this.log(LogLevel.WARN, message, meta);
  }
  
  /**
   * 错误日志
   */
  error(message: string, meta?: any): void {
    this.log(LogLevel.ERROR, message, meta);
  }
  
  /**
   * 设置日志级别
   */
  setLevel(level: LogLevel): void {
    this.level = level;
  }
  
  /**
   * 查询日志
   */
  async query(options: LogQueryOptions): Promise<LogEntry[]> {
    const {
      level,
      module,
      startTime,
      endTime,
      limit = 100,
      offset = 0,
    } = options;
    
    const entries: LogEntry[] = [];
    const files = await this.getLogFiles();
    
    for (const file of files) {
      const content = await fs.readFile(file, 'utf-8');
      const lines = content.split('\n').filter(line => line.trim());
      
      for (const line of lines) {
        try {
          const entry = JSON.parse(line) as LogEntry;
          
          // 过滤条件
          if (level !== undefined && entry.level !== level) continue;
          if (module !== undefined && entry.module !== module) continue;
          if (startTime && new Date(entry.timestamp) < startTime) continue;
          if (endTime && new Date(entry.timestamp) > endTime) continue;
          
          entries.push(entry);
        } catch {
          // 跳过无效行
        }
      }
    }
    
    // 按时间倒序排序
    entries.sort((a, b) => 
      new Date(b.timestamp).getTime() - new Date(a.timestamp).getTime()
    );
    
    // 分页
    return entries.slice(offset, offset + limit);
  }
  
  /**
   * 导出日志
   */
  async export(filePath: string): Promise<void> {
    const files = await this.getLogFiles();
    const content: string[] = [];
    
    for (const file of files) {
      const fileContent = await fs.readFile(file, 'utf-8');
      content.push(fileContent);
    }
    
    await fs.writeFile(filePath, content.join('\n'), 'utf-8');
  }
  
  /**
   * 清空日志
   */
  async clear(): Promise<void> {
    if (this.writeStream) {
      this.writeStream.end();
      this.writeStream = null;
    }
    
    const files = await this.getLogFiles();
    for (const file of files) {
      await fs.unlink(file);
    }
    
    this.currentSize = 0;
    this.rotateLog();
  }
  
  /**
   * 核心日志方法
   */
  private log(level: LogLevel, message: string, meta?: any): void {
    if (level < this.level) {
      return;
    }
    
    const entry: LogEntry = {
      timestamp: new Date().toISOString(),
      level: LogLevel[level],
      module: this.getCallerModule(),
      message,
      meta,
    };
    
    // 错误日志添加堆栈跟踪
    if (level === LogLevel.ERROR && meta instanceof Error) {
      entry.stack = meta.stack;
      entry.meta = { message: meta.message };
    }
    
    const logLine = JSON.stringify(entry) + '\n';
    
    // 检查是否需要轮转
    if (this.currentSize + logLine.length > this.maxSize) {
      this.rotateLog();
    }
    
    // 写入文件
    if (!this.writeStream) {
      this.writeStream = fs.createWriteStream(this.currentFile, { flags: 'a' });
    }
    
    this.writeStream.write(logLine);
    this.currentSize += logLine.length;
    
    // 同时输出到控制台（开发环境）
    if (process.env.NODE_ENV === 'development') {
      this.printToConsole(entry);
    }
  }
  
  /**
   * 日志轮转
   */
  private rotateLog(): void {
    const timestamp = new Date().toISOString().split('T')[0];
    this.currentFile = path.join(this.logDir, `app.${timestamp}.log`);
    this.currentSize = 0;
    
    if (this.writeStream) {
      this.writeStream.end();
      this.writeStream = null;
    }
    
    // 清理旧日志
    this.cleanupOldLogs();
  }
  
  /**
   * 获取日志文件列表
   */
  private async getLogFiles(): Promise<string[]> {
    const files = await fs.readdir(this.logDir);
    return files
      .filter(f => f.endsWith('.log'))
      .map(f => path.join(this.logDir, f))
      .sort()
      .reverse();
  }
  
  /**
   * 清理旧日志
   */
  private async cleanupOldLogs(): Promise<void> {
    const files = await this.getLogFiles();
    
    if (files.length > this.maxFiles) {
      const toDelete = files.slice(this.maxFiles);
      for (const file of toDelete) {
        await fs.unlink(file);
      }
    }
  }
  
  /**
   * 获取调用模块
   */
  private getCallerModule(): string {
    const error = new Error();
    const stack = error.stack?.split('\n') || [];
    const callerLine = stack[3];  // log -> getCallerModule -> caller
    const match = callerLine?.match(/docs\\(.+?)\.ts/);
    return match ? match[1] : 'unknown';
  }
  
  /**
   * 控制台输出
   */
  private printToConsole(entry: LogEntry): void {
    const colors = {
      [LogLevel.DEBUG]: '\x1b[36m',  // Cyan
      [LogLevel.INFO]: '\x1b[32m',   // Green
      [LogLevel.WARN]: '\x1b[33m',   // Yellow
      [LogLevel.ERROR]: '\x1b[31m',  // Red
    };
    
    const color = colors[entry.level] || '\x1b[0m';
    const reset = '\x1b[0m';
    
    console.log(
      `${color}[${entry.timestamp}] [${entry.level}] [${entry.module}]${reset}`,
      entry.message,
      entry.meta || ''
    );
  }
  
  /**
   * 确保日志目录存在
   */
  private async ensureLogDir(): Promise<void> {
    try {
      await fs.access(this.logDir);
    } catch {
      await fs.mkdir(this.logDir, { recursive: true });
    }
  }
}
```

```typescript
// DeviceController.ts
class DeviceController {
  private logger = Logger.getInstance();

    try {
```

## 8. 异常处理

### 8.1 异常类型

| 异常 | 触发条件 | 处理方式 |
|------|----------|----------|
| DiskFull | 磁盘空间不足 | 停止写入，告警 |
| PermissionDenied | 无写入权限 | 尝试创建目录，失败后告警 |
| CorruptedLog | 日志文件损坏 | 跳过损坏行，继续处理 |

## 9. 性能优化

### 9.1 异步写入

```typescript
// 使用写缓冲区，批量写入
class BufferedWriter {
  private buffer: string[] = [];
  private flushTimer: NodeJS.Timeout | null = null;
  private readonly FLUSH_INTERVAL = 1000;  // 1 秒
  private readonly BUFFER_SIZE = 100;      // 100 条
  
  write(line: string): void {
    this.buffer.push(line);
    
    if (this.buffer.length >= this.BUFFER_SIZE) {
      this.flush();
    } else if (!this.flushTimer) {
      this.flushTimer = setTimeout(() => this.flush(), this.FLUSH_INTERVAL);
    }
  }
  
  private flush(): void {
    if (this.buffer.length === 0) return;
    
    const content = this.buffer.join('');
    this.buffer = [];
    
    if (this.flushTimer) {
      clearTimeout(this.flushTimer);
      this.flushTimer = null;
    }
    
    // 异步写入
    fs.appendFile(this.currentFile, content, () => {});
  }
}
```

### 9.2 日志采样

```typescript
// 对 DEBUG 日志进行采样，减少日志量
class SampledLogger extends Logger {
  private sampleRate = 0.1;  // 10% 采样
  
  debug(message: string, meta?: any): void {
    if (Math.random() > this.sampleRate) {
      return;  // 跳过
    }
    super.debug(message, meta);
  }
}
```
