## 2. 模块概述

### 2.1 职责

批处理脚本管理模块负责发现、注册、执行和监控 Windows 上的 Python 批处理脚本，支持 OpenClaw Agent 调用。

### 2.2 功能范围

| 功能 | 说明 | 优先级 |
|------|------|--------|
| 脚本发现 | 扫描脚本目录中的 Python 脚本 | P0 |
| 脚本注册 | 注册脚本到 OpenClaw | P0 |
| 脚本执行 | 手动触发脚本执行 | P0 |
| 状态监控 | 监控脚本执行状态 | P1 |
| 脚本编辑 | 打开脚本编辑器 | P1 |
| 日志查看 | 查看脚本执行日志 | P1 |

### 3.2 核心类说明

#### 3.2.1 BatchScriptManager

批处理脚本管理器。

```typescript
class BatchScriptManager {
  private rootDir: string;
  private scripts: Map<string, BatchScript> = new Map();
  private openclawClient: OpenClawClient;
  private processRunner: ProcessRunner;
  private config: BatchScriptConfig;
  
  constructor(config: BatchScriptConfig, openclawClient: OpenClawClient) {
    this.config = config;
    this.rootDir = config.rootDir;
    this.openclawClient = openclawClient;
    this.processRunner = new ProcessRunner({ timeout: config.timeout || 30000 });
    
    // 确保脚本目录存在
    this.ensureScriptDir();
  }
  
  /**
   * 扫描脚本目录
   */
  async scanScripts(): Promise<BatchScript[]> {
    const files = await fs.readdir(this.rootDir);
    const scripts: BatchScript[] = [];
    
    for (const file of files) {
      if (file.endsWith('.py')) {
        const script = await this.loadScript(file);
        if (script) {
          scripts.push(script);
          this.scripts.set(script.id, script);
        }
      }
    }
    
    return scripts;
  }
  
  /**
   * 加载单个脚本
   */
  private async loadScript(filename: string): Promise<BatchScript | null> {
    const filePath = path.join(this.rootDir, filename);
    
    try {
      // 解析脚本元数据
      const content = await fs.readFile(filePath, 'utf-8');
      const metadata = this.parseScriptMetadata(content, filename);
      
      // 检查是否已注册到 OpenClaw
      const registered = await this.checkRegistration(metadata.name);
      
      return {
        id: generateScriptId(filename),
        name: metadata.name,
        filePath,
        description: metadata.description || '',
        parameters: metadata.parameters || [],
        registered,
        enabled: true,
        executionCount: 0,
      };
    } catch (error) {
      console.error(`Failed to load script ${filename}:`, error);
      return null;
    }
  }
  
  /**
   * 解析脚本元数据
   * 支持从脚本注释中提取元数据
   */
  private parseScriptMetadata(content: string, filename: string): ScriptMetadata {
    const name = path.basename(filename, '.py');
    const descriptionMatch = content.match(/"""([^"]+)"""/);
    const paramsMatch = content.match(/# @param\s+(\w+)\s+(\w+)(?:\s+(.*))?/g);
    
    const parameters: ScriptParameter[] = [];
    if (paramsMatch) {
      for (const line of paramsMatch) {
        const [, type, paramName, desc] = line.match(/# @param\s+(\w+)\s+(\w+)(?:\s+(.*))?/)!;
        parameters.push({
          name: paramName,
          type: type as 'string' | 'number' | 'boolean',
          required: true,
          description: desc,
        });
      }
    }
    
    return {
      name,
      description: descriptionMatch ? descriptionMatch[1].trim() : '',
      parameters,
    };
  }
  
  /**
   * 注册脚本到 OpenClaw
   */
  async registerScript(scriptId: string): Promise<void> {
    const script = this.scripts.get(scriptId);
    if (!script) {
      throw new Error(`Script ${scriptId} not found`);
    }
    
    // 构建工具定义
    const toolDefinition = {
      name: script.name,
      description: script.description,
      command: `python "${script.filePath}"`,
      parameters: script.parameters.map(p => ({
        name: p.name,
        type: p.type,
        required: p.required,
      })),
    };
    
    // 注册到 OpenClaw
    await this.openclawClient.registerTool(toolDefinition);
    
    // 更新本地状态
    script.registered = true;
    await this.saveScriptConfig(script);
  }
  
  /**
   * 从 OpenClaw 注销脚本
   */
  async unregisterScript(scriptId: string): Promise<void> {
    const script = this.scripts.get(scriptId);
    if (!script) {
      throw new Error(`Script ${scriptId} not found`);
    }
    
    await this.openclawClient.unregisterTool(script.name);
    script.registered = false;
    await this.saveScriptConfig(script);
  }
  
  /**
   * 执行脚本
   */
  async executeScript(
    scriptId: string, 
    args?: Record<string, any>
  ): Promise<ExecutionResult> {
    const script = this.scripts.get(scriptId);
    if (!script) {
      throw new Error(`Script ${scriptId} not found`);
    }
    
    // 构建命令行参数
    const cliArgs = this.buildCliArgs(script.parameters, args);
    
    // 执行脚本
    const startTime = Date.now();
    const result = await this.processRunner.run(script.filePath, cliArgs);
    const duration = Date.now() - startTime;
    
    // 更新执行记录
    script.lastExecuted = new Date();
    script.executionCount++;
    await this.saveScriptConfig(script);
    
    // 记录执行历史
    await this.recordExecution(scriptId, result, duration);
    
    return result;
  }
  
  /**
   * 构建命令行参数
   */
  private buildCliArgs(
    parameters: ScriptParameter[], 
    args?: Record<string, any>
  ): string[] {
    const cliArgs: string[] = [];
    
    for (const param of parameters) {
      const value = args?.[param.name] ?? param.defaultValue;
      if (value !== undefined) {
        cliArgs.push(`--${param.name}`, String(value));
      }
    }
    
    return cliArgs;
  }
  
  /**
   * 获取所有脚本
   */
  getAllScripts(): BatchScript[] {
    return Array.from(this.scripts.values());
  }
  
  /**
   * 获取脚本
   */
  getScript(scriptId: string): BatchScript | null {
    return this.scripts.get(scriptId) || null;
  }
  
  /**
   * 获取执行历史
   */
  getExecutionHistory(scriptId: string): ExecutionRecord[] {
    // 从数据库或文件读取
    return this.executionHistory.get(scriptId) || [];
  }
  
  /**
   * 删除脚本
   */
  async deleteScript(scriptId: string): Promise<void> {
    const script = this.scripts.get(scriptId);
    if (!script) {
      throw new Error(`Script ${scriptId} not found`);
    }
    
    // 如果已注册，先注销
    if (script.registered) {
      await this.unregisterScript(scriptId);
    }
    
    // 删除文件
    await fs.unlink(script.filePath);
    
    // 从内存中移除
    this.scripts.delete(scriptId);
  }
  
  /**
   * 确保脚本目录存在
   */
  private async ensureScriptDir(): Promise<void> {
    try {
      await fs.access(this.rootDir);
    } catch {
      await fs.mkdir(this.rootDir, { recursive: true });
    }
  }
}
```

#### 3.2.2 ProcessRunner

进程执行器。

```typescript
class ProcessRunner {
  private timeout: number;
  private runningProcesses: Map<number, ProcessInfo> = new Map();
  
  constructor(options: { timeout: number }) {
    this.timeout = options.timeout;
  }
  
  async run(scriptPath: string, args: string[]): Promise<ExecutionResult> {
    return new Promise((resolve, reject) => {
      const process = spawn('python', [scriptPath, ...args], {
        cwd: path.dirname(scriptPath),
        stdio: ['pipe', 'pipe', 'pipe'],
      });
      
      let stdout = '';
      let stderr = '';
      
      // 设置超时
      const timer = setTimeout(() => {
        process.kill();
        reject(new Error(`Script execution timeout after ${this.timeout}ms`));
      }, this.timeout);
      
      process.stdout.on('data', (data) => {
        stdout += data.toString();
      });
      
      process.stderr.on('data', (data) => {
        stderr += data.toString();
      });
      
      process.on('close', (code) => {
        clearTimeout(timer);
        resolve({
          success: code === 0,
          exitCode: code || 0,
          stdout,
          stderr,
          duration: 0,
        });
      });
      
      process.on('error', (error) => {
        clearTimeout(timer);
        resolve({
          success: false,
          exitCode: -1,
          stdout: '',
          stderr: error.message,
          duration: 0,
        });
      });
    });
  }
  
  async kill(pid: number): Promise<void> {
    // Windows 上使用 taskkill
    execSync(`taskkill /F /PID ${pid}`);
  }
}
```

### 3.3 数据结构

#### 3.3.1 BatchScript

```typescript
interface BatchScript {
  id: string;                   // 脚本 ID
  name: string;                 // 脚本名称
  filePath: string;             // 文件路径
  description: string;          // 描述
  parameters: ScriptParameter[]; // 参数定义
  registered: boolean;          // 是否已注册到 OpenClaw
  enabled: boolean;             // 是否启用
  lastExecuted?: Date;          // 最后执行时间
  executionCount: number;       // 执行次数
}
```

#### 3.3.2 ScriptParameter

```typescript
interface ScriptParameter {
  name: string;                 // 参数名
  type: 'string' | 'number' | 'boolean';
  required: boolean;            // 是否必填
  defaultValue?: any;           // 默认值
  description?: string;         // 描述
}
```

#### 3.3.3 ExecutionResult

```typescript
interface ExecutionResult {
  success: boolean;             // 是否成功
  exitCode: number;             // 退出码
  stdout: string;               // 标准输出
  stderr: string;               // 错误输出
  duration: number;             // 执行耗时 (ms)
}
```

#### 3.3.4 ExecutionRecord

```typescript
interface ExecutionRecord {
  scriptId: string;
  timestamp: Date;
  args?: Record<string, any>;
  result: ExecutionResult;
  triggeredBy: 'user' | 'agent' | 'schedule';
}
```

## 6. 脚本规范

### 6.1 脚本模板

```python
#!/usr/bin/env python3
"""
回家模式 - 开灯、开空调、开电视

执行后会自动打开客厅灯、卧室灯、空调和电视。
"""

# @param string room 房间名称 (可选)
# @param boolean verbose 是否详细输出 (可选，默认 False)

import sys
import argparse
from miio import Device

def parse_args():
    parser = argparse.ArgumentParser(description='回家模式脚本')
    parser.add_argument('--room', type=str, default='客厅', help='房间名称')
    parser.add_argument('--verbose', action='store_true', help='详细输出')
    return parser.parse_args()

def main():
    args = parse_args()
    
    # 设备配置
    devices = {
        "客厅灯": Device("192.168.1.101", "token_xxx"),
        "卧室灯": Device("192.168.1.102", "token_xxx"),
        "空调": Device("192.168.1.103", "token_xxx"),
        "电视": Device("192.168.1.104", "token_xxx"),
    }
    
    results = []
    for name, dev in devices.items():
        try:
            dev.send("set_power", ["on"])
            results.append(f"✅ {name}: 已打开")
            if args.verbose:
                print(f"Successfully turned on {name}")
        except Exception as e:
            results.append(f"❌ {name}: 失败 - {e}")
    
    print("\n".join(results))
    return 0 if all("✅" in r for r in results) else 1

if __name__ == "__main__":
    sys.exit(main())
```

## 7. 异常处理

### 7.1 异常类型

| 异常 | 触发条件 | 处理方式 |
|------|----------|----------|
| ScriptNotFound | 脚本不存在 | 返回错误 |
| ScriptExecutionFailed | 执行失败 | 捕获输出，返回错误 |
| TimeoutError | 执行超时 | 终止进程，返回超时错误 |
